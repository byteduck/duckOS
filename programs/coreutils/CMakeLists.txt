function(MAKE_COREUTIL PROGNAME)
    SET(SOURCES ${PROGNAME}.cpp)
    MAKE_PROGRAM(${PROGNAME})
endfunction()

MAKE_COREUTIL(apprun)
TARGET_LINK_LIBRARIES(apprun libapp)
